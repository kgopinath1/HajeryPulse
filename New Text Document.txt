workflows:
  android-debug:
    name: Android Debug Build
    max_build_duration: 30
    environment:
      node: 20
      vars:
        PROJECT_PATH: mobile
    scripts:
      - name: Install dependencies
        script: |
          cd $PROJECT_PATH
          npm install
      - name: Build Android debug APK
        script: |
          cd $PROJECT_PATH/android
          ./gradlew assembleDebug
    artifacts:
      - mobile/android/app/build/outputs/**/*.apk

  ios-release:
    name: iOS Release Build
    max_build_duration: 60
    environment:
      node: 20
      vars:
        PROJECT_PATH: mobile
      ios_signing:
        # TODO: set up code signing in Codemagic's UI first —
        # Team ID / Bundle ID / distribution type — under this
        # workflow's "Code signing" tab, or via App Store Connect
        # API key for automatic signing. Build WILL fail without this.
        distribution_type: app_store
        bundle_identifier: com.hajerypulse
    triggering:
      events:
        - push
    scripts:
      - name: Install npm dependencies
        script: |
          cd $PROJECT_PATH
          npm install
      - name: Install CocoaPods dependencies
        script: |
          cd $PROJECT_PATH/ios
          pod install
      - name: Set up code signing
        script: |
          keychain initialize
          app-store-connect fetch-signing-files "$BUNDLE_ID" \
            --type IOS_APP_STORE \
            --create
          keychain add-certificates
          xcode-project use-profiles
      - name: Build ipa for distribution
        script: |
          cd $PROJECT_PATH
          xcode-project build-ipa \
            --workspace ios/HajeryPulse.xcworkspace \
            --scheme HajeryPulse
    artifacts:
      - mobile/build/ios/ipa/*.ipa
    publishing:
      email:
        recipients:
          - your-email@hajery.com
